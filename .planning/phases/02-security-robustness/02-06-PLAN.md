class TaskCreateInput(BaseModel):
    """Validation model for task creation input."""

    model_config = ConfigDict(frozen=True)

    name: str
    task_type: int
    data: str
    desc: str | None = None

    @field_validator("name")
    @classmethod
    def validate_name(cls, v: str) -> str:
        if not v or not v.strip():
            raise ValueError("name cannot be empty")
        return v.strip()

    @field_validator("task_type")
    @classmethod
    def validate_task_type(cls, v: int) -> int:
        if v < 0:
            raise ValueError("task_type must be non-negative")
        return v


class ProgressUpdateInput(BaseModel):
    """Validation model for progress update input."""

    model_config = ConfigDict(frozen=True)

    progress: int

    @field_validator("progress")
    @classmethod
    def validate_progress(cls, v: int) -> int:
        if not 0 <= v <= 100:
            raise ValueError("progress must be between 0 and 100")
        return v
```

def main():
    """程序入口点"""
    try:
        app()
    except ValidationError as e:
        rprint(f"[red]输入验证失败: {e}[/red]")
        return 1
    except typer.Exit:
        # typer.Exit is raised by sys.exit() within commands, let it propagate
        raise
    except KeyboardInterrupt:
        rprint("[yellow]操作已取消[/yellow]")
        return 130  # Standard exit code for Ctrl+C
    except Exception as e:
        # Log unexpected errors with full traceback for debugging
        rprint(f"[red]程序异常: {e!s}[/red]")
        return 1
    return 0
```
