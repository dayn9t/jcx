def check(self, clock_time: ClockTime) -> bool:
    """判定时间是否满足日历触发条件"""
    # 时段检查
    if self.periods:
        ok = any(clock_time in p for p in self.periods)
        if not ok:
            return False

    # 星期检查
    if self.weekdays is not None:
        # Convert clock_time to get weekday
        # Assuming ClockTime has a way to get datetime or weekday
        dt = clock_time.to_datetime()  # Adjust based on actual ClockTime API
        if dt.weekday() not in self.weekdays:
            return False

    return True
