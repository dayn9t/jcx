<!-- Key types from rustshed used for error handling -->
```python
from rustshed import Result, Ok, Err, Option, Some, Null

# Pattern: Return Result to caller
def fallible_operation() -> Result[T, E]:
    result = some_operation()
    if result.is_err():
        return Err(result.unwrap_err())
    return Ok(result.unwrap())

# Pattern: unwrap_or for defaults
value = optional_value.unwrap_or(default_value)

# Pattern: expect for unrecoverable errors (when signature cannot change)
value = result.expect("Clear message about what went wrong")
```
